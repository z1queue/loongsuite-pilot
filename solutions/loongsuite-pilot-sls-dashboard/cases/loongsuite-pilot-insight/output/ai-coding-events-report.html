  SELECT
    date_trunc('day', __time__) AS t,
    "user.id" AS user_id,
    coalesce(nullif("gen_ai.agent.type", 'null'), nullif("gen_ai.agent.name", 'null'), 'unknown') AS agent_type,
    coalesce(nullif("gen_ai.provider.name", 'null'), 'unknown') AS provider,
    coalesce(nullif("gen_ai.request.model", 'null'), nullif("gen_ai.response.model", 'null'), 'unknown') AS model,
    sum(coalesce("gen_ai.usage.input_tokens", 0)) AS input_tokens,
    sum(coalesce("gen_ai.usage.output_tokens", 0)) AS output_tokens,
    sum(coalesce("gen_ai.usage.total_tokens", 0)) AS total_tokens,
    count(1) AS events

  GROUP BY t, user_id, agent_type, provider, model
),
user_token AS (
  SELECT
    user_id,
    sum(total_tokens) AS total_tokens
  FROM event_user_day
  GROUP BY user_id
  HAVING total_tokens > 0
),
ranked AS (
  SELECT
    user_id,
    total_tokens,
    row_number() OVER (ORDER BY total_tokens DESC) AS rn,
    count(1) OVER () AS user_count,
    sum(total_tokens) OVER () AS all_tokens
  FROM user_token
)
SELECT round(sum(CASE WHEN rn <= cast(ceil(user_count * 0.10) AS bigint) THEN total_tokens ELSE 0 END) * 100.0 / nullif(max(all_tokens), 0), 2) AS "Top10% Token占比"
FROM ranked

WITH event_user_day AS (
  SELECT
    date_trunc('day', __time__) AS t,
    "user.id" AS user_id,
    coalesce(nullif("gen_ai.agent.type", 'null'), nullif("gen_ai.agent.name", 'null'), 'unknown') AS agent_type,
    coalesce(nullif("gen_ai.provider.name", 'null'), 'unknown') AS provider,
    coalesce(nullif("gen_ai.request.model", 'null'), nullif("gen_ai.response.model", 'null'), 'unknown') AS model,
    sum(coalesce("gen_ai.usage.input_tokens", 0)) AS input_tokens,
    sum(coalesce("gen_ai.usage.output_tokens", 0)) AS output_tokens,
    sum(coalesce("gen_ai.usage.total_tokens", 0)) AS total_tokens,
    count(1) AS events
  FROM log
  WHERE "user.id" IS NOT NULL AND "user.id" <> ''
  GROUP BY t, user_id, agent_type, provider, model
),
user_token AS (
  SELECT
    user_id,
    sum(total_tokens) AS total_tokens
  FROM event_user_day
  GROUP BY user_id
  HAVING total_tokens > 0
),
ranked AS (
  SELECT
    user_id,
    total_tokens,
    row_number() OVER (ORDER BY total_tokens DESC) AS rn,
    count(1) OVER () AS user_count,
    sum(total_tokens) OVER () AS all_tokens
  FROM user_token
)
SELECT round(sum(CASE WHEN rn <= cast(ceil(user_count * 0.20) AS bigint) THEN total_tokens ELSE 0 END) * 100.0 / nullif(max(all_tokens), 0), 2) AS "Top20% Token占比"
FROM ranked

WITH event_user_day AS (
  SELECT
    date_trunc('day', __time__) AS t,
    "user.id" AS user_id,
    coalesce(nullif("gen_ai.agent.type", 'null'), nullif("gen_ai.agent.name", 'null'), 'unknown') AS agent_type,
    coalesce(nullif("gen_ai.provider.name", 'null'), 'unknown') AS provider,
    coalesce(nullif("gen_ai.request.model", 'null'), nullif("gen_ai.response.model", 'null'), 'unknown') AS model,
    sum(coalesce("gen_ai.usage.input_tokens", 0)) AS input_tokens,
    sum(coalesce("gen_ai.usage.output_tokens", 0)) AS output_tokens,
    sum(coalesce("gen_ai.usage.total_tokens", 0)) AS total_tokens,
    count(1) AS events
  FROM log
  WHERE "user.id" IS NOT NULL AND "user.id" <> ''
  GROUP BY t, user_id, agent_type, provider, model
),
user_token AS (
  SELECT
    user_id,
    sum(total_tokens) AS total_tokens
  FROM event_user_day
  GROUP BY user_id
  HAVING total_tokens > 0
),
ranked AS (
  SELECT
    user_id,
    total_tokens,
    row_number() OVER (ORDER BY total_tokens DESC) AS rn,
    count(1) OVER () AS user_count,
    sum(total_tokens) OVER () AS all_tokens
  FROM user_token
),
layered AS (
  SELECT
    CASE
        WHEN rn <= cast(ceil(user_count * 0.10) AS bigint)
        THEN 'Top 10%'
        WHEN rn <= cast(ceil(user_count * 0.20) AS bigint)
        THEN '10%-20%'
        ELSE '后80%'
      END AS user_layer,
    CASE
        WHEN rn <= cast(ceil(user_count * 0.10) AS bigint)
        THEN 1
        WHEN rn <= cast(ceil(user_count * 0.20) AS bigint)
        THEN 2
        ELSE 3
      END AS layer_rank,
    total_tokens,
    all_tokens
  FROM ranked
)
SELECT
  user_layer AS "人群分层",
  count(1) AS "人数",
  coalesce(sum(total_tokens), 0) AS "Token",
  round(sum(total_tokens) * 100.0 / nullif(max(all_tokens), 0), 2) AS "Token占比(%)"
FROM layered
GROUP BY user_layer, layer_rank
ORDER BY layer_rank

WITH event_user_day AS (
  SELECT
    date_trunc('day', __time__) AS t,
    "user.id" AS user_id,
    coalesce(nullif("gen_ai.agent.type", 'null'), nullif("gen_ai.agent.name", 'null'), 'unknown') AS agent_type,
    coalesce(nullif("gen_ai.provider.name", 'null'), 'unknown') AS provider,
    coalesce(nullif("gen_ai.request.model", 'null'), nullif("gen_ai.response.model", 'null'), 'unknown') AS model,
    sum(coalesce("gen_ai.usage.input_tokens", 0)) AS input_tokens,
    sum(coalesce("gen_ai.usage.output_tokens", 0)) AS output_tokens,
    sum(coalesce("gen_ai.usage.total_tokens", 0)) AS total_tokens,
    count(1) AS events
  FROM log
  WHERE "user.id" IS NOT NULL AND "user.id" <> ''
  GROUP BY t, user_id, agent_type, provider, model
),
daily_user AS (
  SELECT
    t,
    user_id,
    sum(total_tokens) AS total_tokens
  FROM event_user_day
  GROUP BY t, user_id
  HAVING total_tokens > 0
),
ranked AS (
  SELECT
    t,
    user_id,
    total_tokens,
    row_number() OVER (PARTITION BY t ORDER BY total_tokens DESC) AS rn,
    count(1) OVER (PARTITION BY t) AS user_count,
    sum(total_tokens) OVER (PARTITION BY t) AS all_tokens
  FROM daily_user
)
SELECT
  t,
  round(sum(CASE WHEN rn <= cast(ceil(user_count * 0.10) AS bigint) THEN total_tokens ELSE 0 END) * 100.0 / nullif(max(all_tokens), 0), 2) AS "Top10% Token占比",
  round(sum(CASE WHEN rn <= cast(ceil(user_count * 0.20) AS bigint) THEN total_tokens ELSE 0 END) * 100.0 / nullif(max(all_tokens), 0), 2) AS "Top20% Token占比"
FROM ranked
GROUP BY t
ORDER BY t
LIMIT 10000
