  FROM user_token
),
layered AS (
  SELECT
    CASE
        WHEN rn <= cast(ceil(user_count * 0.10) AS bigint)
        THEN 'Top 10%'
        WHEN rn <= cast(ceil(user_count * 0.20) AS bigint)
        THEN '10%-20%'
        ELSE '后80%'
      END AS user_layer,
    CASE

    all_tokens
  FROM ranked

FROM layered
GROUP BY user_layer, layer_rank
